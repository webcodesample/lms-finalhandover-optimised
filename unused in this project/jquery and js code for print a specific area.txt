function printTable() {
    $("center").addClass("border border-danger py-5 vh-100")
    var divName = "myTable"; // Replace with your table's div ID
    var printContents = $("#myTable").html();
    var originalContents = $("body").html();
    $("body").html(printContents);
    window.print();
    $("body").html(originalContents); // Restore the original content
    $("center").removeClass("border border-danger py-5 vh-100")
}

function printTable2() {
